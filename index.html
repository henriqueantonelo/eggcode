Plus Minus

Given an array of integers, calculate the ratios of its elements that are positive, negative, and zero. Print the decimal value of each fraction on a new line with places after the decimal. 

          

            const plusMinus = function (arr) {
              let positiveArr = [];
              let negativeArr = [];
              let zeroArr = [];
              for (let i = 0; i < arr.length; i++) {
                if (arr[i] < 0) {
                  negativeArr.push(arr[i]);
                } else if (arr[i] > 0) {
                  positiveArr.push(arr[i]);
                } else {
                  zeroArr.push(arr[i]);
                }
              }
              let positive = (positiveArr.length / arr.length).toFixed(6);
              let negative = (negativeArr.length / arr.length).toFixed(6);
              let zero = (zeroArr.length / arr.length).toFixed(6);
              console.log(positive, negative, zero);
            };
            
            plusMinus([-4, 3, -9, 0, 4, 1]);

Mini-Max Sum

Given five positive integers, find the minimum and maximum values that can be calculated by summing exactly four of the five integers. Then print the respective minimum and maximum values as a single line of two space-separated long integers. 

          

            const miniMaxSum = function (arr) {
              arr.sort();
              let lowest = arr.shift();
              let highest = arr.pop();
              const maxSum = function () {
                let maxArr = 0;
                maxArr = arr;
                maxArr.push(highest);
                maxArr.sort();
                let sum = 0;
                for (let i = 0; i < maxArr.length; i++) {
                  sum += arr[i];
                }
                return sum;
              };
            
              const miniSum = function () {
                let minArr = 0;
                minArr = arr;
                minArr.push(lowest);
                minArr.sort();
                minArr.pop();
                let sum = 0;
                for (let i = 0; i < arr.length; i++) {
                  sum += arr[i];
                }
                return sum;
              };
            
              let maxSumResult = maxSum();
              let miniSumResult = miniSum();
              console.log(miniSumResult, maxSumResult);
            };
            
            miniMaxSum([7, 69, 2, 221, 8974]);

Pangrams

A pangram is a string that contains every letter of the alphabet. Given a sentence determine whether it is a pangram in the English alphabet. Ignore case. Return either pangram or not pangram as appropriate. 

          


      const pangram = function (s) {
        const alphabet = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
        let inputUppercase = s.toUpperCase();
      
        function removeDuplicate() {
          let charArray = inputUppercase.split("");
          let uniqueChars = [...new Set(charArray)];
          uniqueChars.sort();
          let inputNoSpecialChar = uniqueChars.toString().replace(/[^A-Z0-9]+/gi, "");
          return inputNoSpecialChar;
        }
      
        function subtractCharacters(str1, str2) {
          let result = str1;
          for (const char of str2) {
            const regex = new RegExp(char, "g");
            result = result.replace(regex, "");
          }
          return result;
        }
      
        function moreThanOne() {
          let more = subtractCharacters(alphabet, removeDuplicate(s));
          const myArray = more.split("");
          return myArray;
        }
      
        moreThanOne();
      
        if (removeDuplicate(s) == alphabet) {
          console.log("It is a pangram!");
        } else {
          console.log(`It's not a pangram, because it's missing ${moreThanOne()}`);
        }
      };

      pangram("We promptly judged antique ivory bt prize");

            function timeConversion(s) {
              let hour = s.slice(0, -8);
              let period = s.slice(-2);
              let minuteSecond = s.slice(2, -2);
              let midnight = "00";
            
              if (period === "AM" && hour <= "11") {
                let am = s.slice(0, -2);
                return am;
              } else if (period === "AM" && hour === "12") {
                let convertion = midnight + minuteSecond;
                return convertion;
              } else if (period === "PM" && hour < "12") {
                let num = Number(hour);
                let convertion = num + 12;
                let pm = convertion.toString() + minuteSecond;
                return pm;
              } else if (period === "PM" && hour === "12") {
                let pm = s.slice(0, -2);
                return pm;
              }
            }
